package lab_3java;

import java.util.GregorianCalendar;

/**
 *
 * @author Greg
 */
public class TestGregCal {

    public static void main(String[] args) {
        GregorianCalendar set = new GregorianCalendar();
        System.out.println("The current year, month and day denoted by the "
                + "Gregorian Calendar are: "
                + set.get(GregorianCalendar.YEAR) + " "
                + set.get(GregorianCalendar.MONTH) + " "
                + set.get(GregorianCalendar.DAY_OF_MONTH));

        /**
         * GregorianCalendar.YEAR + " " + GregorianCalendar.MONTH + " " +
         * GregorianCalendar.DAY_OF_MONTH );
         */
        set.setTimeInMillis(1234567898765L);
        System.out.println("The set value since January 1, 1970 (linux time) is: "
                + set.get(GregorianCalendar.YEAR) + " "
                + set.get(GregorianCalendar.MONTH) + " "
                + set.get(GregorianCalendar.DAY_OF_MONTH));
    }
}
